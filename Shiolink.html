def add_line(line)
  unless @session
    case line[0 ... 3]
    when '*L:'
      path = line[3 ... line.size].chomp + line.match(/[\/\\]/).to_s
      @load.call(path)
      nil
    when '*S:'
      @session = ''
      line
    when '*U:'
      @unload.call
      nil
    end
  else
    @session += line
    if line.chomp.size != 0
      nil
    else
      response = @request.call(@session)
      @session = nil
      response
    end
  end
end

def start(input = $stdin, output = $stdout)
  input.each_line do |line|
    if response = add_line(line)
      output.print(response)
      #output.flush
    end
  end
end
